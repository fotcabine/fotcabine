App.js

import React, { useState } from 'react';
import './App.css'; // Estilos gerais
import 'react-image-lightbox/style.css'; // Estilo do lightbox

import ReactLightbox from 'react-image-lightbox'; // Importando o Lightbox

function App() {
  // Gerar as imagens para a galeria (foto10, foto11, ..., foto65)
  const galeriaImages = [];
  for (let i = 10; i <= 65; i++) {
    galeriaImages.push(`/imagens/foto${i}.jpg`);
  }

  // Estado para controlar o lightbox
  const [isOpen, setIsOpen] = useState(false);
  const [photoIndex, setPhotoIndex] = useState(0);

  // Função para abrir o lightbox
  const openLightbox = (index) => {
    setIsOpen(true);
    setPhotoIndex(index);
  };

  return (
    <>
      {/* Seção A Fotocabine */}
      <section id="fotocabine" className="py-20 bg-gray-100 text-black">
        <div className="container mx-auto text-center">
          <h2 className="text-4xl mb-6 text-laranja">A Fotocabine</h2>
          <p className="text-justify max-w-3xl mx-auto">
            Há 11 anos no mercado a Fotcabine teve o privilégio de estas presente nos momentos mais emocionantes como casamentos, formaturas, 15 anos e eventos corporativos de vários tipos.  Temos orgulho em dizer que nunca deixamos de cumprir nossos contratos. Com equipamentos tradicionais como a CABINE DE FOTOS, sofisticados como o ESPELHO MÁGICO e como não poderia faltar os modernos como PLATAFORMA ou MOBILEBOOTH, estamos sempre criando recordações para uma vida toda, pois levamos para sua festa Mais que uma lembrança!
          </p>
        </div>
      </section>

      {/* Seção Galeria */}
      <section id="galeria" className="bg-gray-100 py-20">
        <div className="container mx-auto text-center">
          <h2 className="text-4xl mb-6 text-laranja">Galeria</h2>
          <div className="grid grid-cols-8 gap-6">
            {/* Exibindo as imagens da galeria */}
            {galeriaImages.map((image, index) => (
              <div key={index} className="bg-blue-300 p-6">
                <img
                  src={image}
                  alt={`Foto ${index + 10}`}
                  className="w-full h-auto object-cover cursor-pointer"
                  onClick={() => openLightbox(index)} // Abre o lightbox ao clicar na imagem
                />
              </div>
            ))}
          </div>
        </div>
      </section>

      {/* Lightbox para exibir a imagem ampliada */}
      {isOpen && (
        <ReactLightbox
          mainSrc={galeriaImages[photoIndex]} // A imagem que está sendo visualizada
          nextSrc={galeriaImages[(photoIndex + 1) % galeriaImages.length]} // Próxima imagem
          prevSrc={galeriaImages[(photoIndex + galeriaImages.length - 1) % galeriaImages.length]} // Imagem anterior
          onCloseRequest={() => setIsOpen(false)} // Fecha o lightbox
          onMovePrevRequest={() =>
            setPhotoIndex((photoIndex + galeriaImages.length - 1) % galeriaImages.length)
          }
          onMoveNextRequest={() => setPhotoIndex((photoIndex + 1) % galeriaImages.length)} // Navegação para próxima
        />
      )}

      {/* Seção Eventos */}
      <section id="eventos" className="py-20 bg-preto text-white">
        <div className="container mx-auto text-center">
          <h2 className="text-4xl mb-6 text-laranja">Próximos Eventos</h2>
          <ul className="list-disc text-left">
            <li>Evento 1 - Data</li>
            <li>Evento 2 - Data</li>
            <li>Evento 3 - Data</li>
          </ul>
        </div>
      </section>
    </>
  );
}

export default App;
